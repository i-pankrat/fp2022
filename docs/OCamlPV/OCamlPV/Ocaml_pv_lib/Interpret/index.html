type 'a binop =
| EmptyBinOp of Ast.bin_op(*

Represents binary operation with no arguments applied: +

*)
| PartialBinOp of 'a * Ast.bin_op(*

Represents binaty operation with one argument applied: 4 +

*)
type value =
| VInt of int(*

Represents integer values: 1, 2, ...

*)
| VBool of bool(*

Represents boolean values: false, true

*)

Represents string values: "Ocaml is cool!"

*)

Represents tuples: (value1, value2, ... )

*)

Represents lists: value1; value2; ...

*)

Represents function that is not applied to its pattern

*)

Represents binaty operation with no arguments applied and one argument applied

*)
